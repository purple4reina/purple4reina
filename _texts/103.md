from itertools import *

def rule1(a):
	for i in range(1, len(a)/2+1):
		for A in combinations(a, i):
			for j in range(1, len(a)-len(A)+1):
				for B in combinations((set(a)-set(A)), j):
					if sum(list(A)) == sum(list(B)):
						return False
	return True
