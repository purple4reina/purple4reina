```python
from time import time
start_time= time()
print 'Process started'

#n  n/2 (n is even)
#n  3n + 1 (n is odd)


def seq(n):
	iter = 1
	while n != 1:
		if n%2 == 1:
			n = (3*n) + 1
		elif n%2 == 0:
			n = n/2
		iter = iter + 1
	return iter
			
seq_list = []
for i in range(1000000, 0, -1):
	seq_list.append([seq(i), i])

print max(seq_list)


end_time= time()
elapse_time= end_time - start_time
print elapse_time, 'seconds have passed'
```
