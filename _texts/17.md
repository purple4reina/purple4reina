	ans = []
	if n == 1000:
		ans.append('one')
		ans.append('thousand')
	
	d1 = n%10
	d2 = (n%100 - d1)/10
	d3 = (n%1000 - d1 - d2)/100
	
	if d3 != 0:
		if d3 == 1:
			ans.append('one')

		ans.append('hundred')
	
	if d3 != 0 and (d2 != 0 or d1 != 0):
		ans.append('and')
	
	if d2 != 0:

			ans.append('nine')

total = 0
