count = 0
bound = 10**6

N = dict(zip(xrange(1, bound+1), [0]*bound))

for outer in range(3, bound/4 + 2):
	if outer%1000 == 0: print outer
	for inner in range(outer-2, 0, -2):
		this = outer**2 - inner**2
		if this <= bound:
			N[this] += 1
		if this > bound:
			break

total = 0
for v in N.itervalues():
	if 1 <= v <= 10:
		total += 1

print total
