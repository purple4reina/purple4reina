def t(n):
    yield 1
    yield 1
    yield 1
    t1 = t2 = t3 = 1
    for _ in range(n-3):
        t1, t2, t3 = t2, t3, t1+t2+t3
        yield t3

def solve(n):
    tribs = tuple(t(10**5))
    count = 0
    k = 1
    while count < n:
        k += 2
        for tr in tribs:
            if tr % k == 0:
                break
        else:
            count += 1
    return k

if __name__ == '__main__':
