def isprime(n):
	for i in range(2, int(n**0.5 + 1)):
		if n%i == 0: return 'FALSE'
	return 'TRUE'

from itertools import count
def longest(a, b):
	for n in count(0):
		quadratic = n**2 + a*n + b 
		if quadratic < 0: return 0
		elif isprime(quadratic) == 'FALSE': return n
