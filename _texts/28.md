```python
bound = 1001

diagonals = [1]
for r in range(1, ((bound-1)/2)+1):
	adder =  2 * r
	for c in range(1, 5):
		next = max(diagonals) + adder
		diagonals.append(next)
		if next == (bound**2): break; break
		
print sum(diagonals)
```
