import re
def num2str(n):
	return re.split('_', re.sub('\B', '_', str(n)))

def str2num(n):
	return ''.join(map(str, n))	

def circle(n):
	return int(str2num([(list(num2str(n)))[-1]] + (list(num2str(n)))[:-1]))
	
def isprime(n):
	for i in range(2, int(n**0.5 + 1)):
		if n%i == 0: return 'FALSE'
	return 'TRUE'
