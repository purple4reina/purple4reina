## Code

```python
triplets = []
perimeters = []
for k in range(1, 340):
	for m in range(1, 40):
		for n in range(1, 40):
			print (k, m, n)
			a = k*(m**2-n**2)
			b = k*2*m*n
			c = k*(m**2+n**2)
			if ((a+b+c) <= 1000) and ((a, b, c) not in triplets) and (a > 0):
				triplets.append((a, b, c))
				perimeters.append(a+b+c)

countlist = []
for n in range(1001):
	countlist.append((perimeters.count(n), n))
	
print max(countlist)
```
