def isprime(n):
	if n == 0 or n == 1: return 'false'
	for i in range(2, int(n**.5) + 1):
		if n%i == 0: return 'false'
	return 'true'

primeslist = []

def factorcount(n):
	times = 0
	for p in primeslist:
		if n%p == 0: times += 1
	return times
