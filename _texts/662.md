## Code

```python
modulo = 1000000007

def fibonacci(top):
    a, b = 1, 1
    while b <= top:
        yield b
        a, b = b, b + a

def solve(warg, harg):
    fibs = tuple(fibonacci(harg * 2))
    grid = {(w, h): 0 for h in range(harg + 1) for w in range(warg + 1)}
    grid[(0, 0)] = 1

    reaches = []
    for h in range(harg + 1):
        for w in range(warg + 1):
            if (w**2 + h**2) ** 0.5 in fibs:
                reaches.append((w, h))

    for (width, height), mult in grid.items():
        mult %= modulo
        for w, h in reaches:
            if height + h > harg:
                break
            if width + w > warg:
                continue
            reached = (width + w, height + h)
            grid[reached] += mult

    return grid[reached] % modulo

if __name__ == '__main__':
    import sys
    n = eval(sys.argv[1])
    m = eval(sys.argv[1])
    print(solve(n,m))
```
