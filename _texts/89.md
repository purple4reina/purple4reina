def value(R):
	total = 0
	for i in range(len(R)):
		d = R[i]
		if d == 'I':
			if 'V' in R[i:] or 'X' in R[i:]:
				total -= 1
			else:
				total += 1
		elif d == 'V':
			total += 5
		elif d == 'X':
			if 'L' in R[i:] or 'C' in R[i:]:
				total -= 10
			else:
				total += 10
		elif d == 'L':
			total += 50
		elif d == 'C':
			if 'D' in R[i:] or 'M' in R[i:]:
				total -= 100
			else:
				total += 100
		elif d == 'D':
			total += 500
		elif d == 'M':
			total += 1000
	return total

	thousand = n%10000 - n%1000
	hundred = n%1000 - n%100
	ten = n%100 - n%10
	one = n%10
