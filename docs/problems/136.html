def solve(n, m):
    solutions = {}
    for i in range(3, 5*m//4+1):
        for j in range(4*i//5, i//2, -1):
            key = 4*i*j - 5*j**2
            if key >= m:
                break
            solutions[key] = solutions.get(key, 0) + 1
    count = 0
    for val in solutions.values():
        count += val == n
    return count

if __name__ == '__main__':
    import sys
    n = eval(sys.argv[1])
    m = eval(sys.argv[2])
    print(solve(n, m))
