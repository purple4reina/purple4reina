def X(a, b, c):
    return a ^ b ^ c

def solve(n):
    total = 0
    for k in range(1, n + 1):
        if not X(k, 2*k, 3*k):
            total += 1
    return total

if __name__ == '__main__':
    n = 1 << 30
    print(solve(n))
