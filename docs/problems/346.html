def solve(top):
    repunits = {}
    for base in range(2, top):
        num = base + 1
        exp = 1
        while num < top:
            repunits[num] = repunits.get(num, 0) + 1
            exp += 1
            num += base ** exp
        if exp == 2:
            break

    return sum(k for k, v in repunits.items() if v > 1 or k > base+1) + 1

if __name__ == '__main__':
    import sys
    n = int(sys.argv[1])
    print(solve(n))
