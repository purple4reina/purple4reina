def cache(fn):
    _cache = {}
    def wrap(n):
        if n in _cache:
            return _cache[n]
        ret = fn(n)
        _cache[n] = ret
        return ret
    return wrap

def solve(n):

    top = int(n ** 0.5)
    sieve = [True] * (top + 1)
    primes = []
    for p, is_prime in enumerate(sieve):
        if not is_prime or p < 2:
            continue
        primes.append(p)
        num = p + p
        while num <= top:
            sieve[num] = False
            num += p

    @cache
    def totient(k):
        if k == 1:
            return 1
        top = k ** 0.5
        for p in primes:
            if p > top:
                break
            if k % p == 0:
                k //= p
                e = 1
                while k % p == 0:
                    k //= p
                    e += 1
                return p ** e * (1 - 1/p) * totient(k)
        return k - 1

    visible = 0
    for r in range(1, n + 1):
        visible += totient(r)

    orchard = n * (n + 1) // 2
    return 6 * int(orchard - visible)

if __name__ == '__main__':
    import sys
    n = eval(sys.argv[1])
    print(solve(n))
