def str2num(n):
	return ''.join(map(str, n))		

#create a concatenation and break if it cannot be 9 digits exactly
from math import log10
def concatenate(n):
	num = (n)
	mult = 2
	while (log10(float(num)) < 8):
		num = str2num((num, mult * n))
		mult += 1
	return num
