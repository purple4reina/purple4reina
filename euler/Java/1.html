public class One {
	
	public static void main(String[] args) {
		int count;
		count = 0;
		
		for (int threeMult = 0; 3*threeMult < 1000; threeMult++)
			count += 3*threeMult;
		
		for (int fiveMult = 0; 5*fiveMult < 1000; fiveMult++)
			count += 5*fiveMult;
		
		for (int bothMult = 0; 15*bothMult < 1000; bothMult++)
			count -= 15*bothMult;
		
			
		System.out.println(count);	
			
			
	}
