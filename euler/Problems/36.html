import re
def num2str(n):
	return re.split('_', re.sub('\B', '_', str(n)))
	
def ispalendrome(n):
	n = num2str(n)
	for i in range(len(n)):
		if n[i] == n[-(i+1)]: continue
		else: return 'FALSE'
	return 'TRUE'

sum = 0
