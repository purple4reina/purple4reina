def divide(x, y):
	x = int(x)
	y = int(y)
	z = []
	while len(z) < 10:
		if (x / y) >= 1:
			z.append(x / y)
			x = x%y * 10
		else:
			x = x*10
			z.append(0)
	return int(''.join(map(str, z[1:])))
