This one was simple. I just made a python script to test each of the different rules.

def tests(n):
	number = [int(n) for n in str(n)]
	assert number[0] * number[1] == 24
	assert number[3] * 2 == number[1]
	assert (number[3] + number[4]) == (number[0] + number[2])
	assert sum(number) == 26
	assert number[1] > number[4]


for n in xrange(10000, 100000):
	try:
		tests(n)
		print n
	except:
		pass
